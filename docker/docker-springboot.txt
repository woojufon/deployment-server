1、安装mvn
MAVEN_HOME=/home/apache-maven-3.5.4
export MAVEN_HOME
export PATH=${PATH}:${MAVEN_HOME}/bin

2、安装java与docker

3、项目配置Dockerfile

FROM 192.168.12.242/spush/docker-alpine-java:1.8
ADD springboot-test-1.0.0.jar /usr/app/
ENV HOME /usr/app
WORKDIR /usr/app/
ENTRYPOINT ["java",  "-jar", "springboot-test-1.0.0.jar"]

4、docker打包提交本地镜像

mvn package docker:build -Dmaven.test.skip=true

5、启动

docker run --name springboot-test  -p 8881:80 -d springboot/springboot-test
